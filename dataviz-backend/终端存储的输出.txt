services:
  dataviz-backend:
    image: dataviz-backend:latest
    container_name: dataviz-backend
    ports:
      - "8000:8000"
    environment:
      - DATABASE_URL=${DATABASE_URL}
      - LLM_API_KEY=${LLM_API_KEY}
      - LLM_BASE_URL=${LLM_BASE_URL}
    volumes:
      - ./dataviz-backend/uploads:/app/uploads
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8000/health"]
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 40s
    networks:
      - dataviz-network

  dataviz-frontend:
    image: dataviz-frontend:latest
    container_name: dataviz-frontend
    ports:
      - "5000:5000"

      - VITE_API_BASE_URL=http://dataviz-backend:8000
    depends_on:
      - dataviz-backend
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "wget", "--quiet", "--tries=1", "--spider", "http://localhost:5000"]
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 40s
    networks:
      - dataviz-network
